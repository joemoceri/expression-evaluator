using System;
using System.Collections.Generic;

namespace ExpressionEvaluatorForDotNet
{
    public class HL7V[{-VERSION-}]Field
    {
        public HL7V2Field field { get; init; }

        public IList<HL7V[{-VERSION-}]FieldRepetition> fieldRepetitions { get; internal set; }

        public HL7V[{-VERSION-}]FieldData fieldData { get; init; }

        public string Value { get { return field?.Value; } }

        public HL7V2ValidationResult Validate()
        {
            var result = HL7V[{-VERSION-}]FieldValidator.ValidateV[{-VERSION-}]Field(this);

            return result;
        }
    }
}
