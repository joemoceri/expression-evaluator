using System.Collections.Generic;

namespace ExpressionEvaluatorForDotNet
{
    public class HL7V[{-VERSION-}]SubComponent
    {
        public HL7V2SubComponent subComponent { get; init; }

        public HL7V[{-VERSION-}]FieldData fieldData { get; init; }

        public string Value { get { return subComponent.Value; } }

        public HL7V2ValidationResult Validate()
        {
            var result = HL7V[{-VERSION-}]SubComponentValidator.ValidateV[{-VERSION-}]SubComponent(this);

            return result;
        }
    }
}
