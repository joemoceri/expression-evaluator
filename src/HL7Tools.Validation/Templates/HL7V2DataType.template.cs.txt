        public string Name { get { return [{-NAME-}]; } }

        public string Description { get { return [{-DESCRIPTION-}]; } }

        public string DataType { get { return [{-DATA_TYPE-}]; } }

        public string DataTypeName { get { return [{-DATA_TYPE_NAME-}]; } }

        public int Length { get { return [{-LENGTH-}]; } }

        public string Usage { get { return [{-USAGE-}]; } }

        public string Rpt { get { return [{-RPT-}]; } }

        public string TableId { get { return [{-TABLE_ID-}]; } }

        public string TableName { get { return [{-TABLE_NAME-}]; } }

        public string Sample { get { return [{-SAMPLE-}]; } }

        public IList<HL7V2FieldData> Fields

            get

                return [{-FIELDS-}];
            }
        }
    }
